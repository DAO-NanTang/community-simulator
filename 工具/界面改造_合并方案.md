```js
var memberFilter = 'builder';

function filterMembers(type) {
  memberFilter = type;
  var types = ['builder','adventurer','npc','visitor'];
  document.querySelectorAll('#memberSubNav .btn').forEach(function(b, i) {
    b.className = (types[i] === type) ? 'btn pri' : 'btn';
  });
  selectedMember = null;
  renderMembers();
}

function renderMembers() {
  var users = getUsers();
  // 从 nt_users 取该类型的所有人
  var filtered = Object.values(users).filter(function(u) { return u.role === memberFilter; });
  
  var avRow = document.getElementById('memberAvatars');
  // 清除旧头像（保留 + 按钮）
  avRow.querySelectorAll('.ms-avatar-btn:not(.ms-add-btn)').forEach(function(b) { b.remove(); });
  var addBtn = avRow.querySelector('.ms-add-btn');
  
  if (filtered.length === 0) {
    document.getElementById('memberDetail').innerHTML = '';
    document.getElementById('memberEmpty').classList.remove('hidden');
    return;
  }
  document.getElementById('memberEmpty').classList.add('hidden');
  
  filtered.forEach(function(u) {
    var mi = data.members && data.members[u.name] ? data.members[u.name] : {};
    var btn = document.createElement('button');
    btn.className = 'ms-avatar-btn' + (u.name === selectedMember ? ' active' : '');
    btn.onclick = function() { selectMember(u.name); };
    btn.ondblclick = function(e) { e.stopPropagation(); showEditMember(u.name); };
    btn.title = '双击编辑';
    var sub = mi.role || '';
    btn.innerHTML = '<div class="ms-av-circle">' + avatarCircle(u.avatar_seed || 0, 64, 0) + '</div><div class="ms-av-name">' + esc(u.name) + '</div><div class="ms-av-badge">' + esc(sub) + '</div>';
    avRow.insertBefore(btn, addBtn);
  });
  
  if (!selectedMember || !filtered.find(function(u) { return u.name === selectedMember; })) {
    selectedMember = filtered[0].name;
  }
  renderMemberDetail(selectedMember);
}
```

```js
var questFilter = '全部';

function renderQuestBoard() {
  var tasks = data.tasks || [];
  var container = document.getElementById('tab-map');
  var name = currentUser ? currentUser.name : '';
  
  // 筛选
  var filtered = tasks.filter(function(t) {
    if (t.status !== 'active') return false;
    if (questFilter !== '全部' && t.type !== questFilter) return false;
    // 隐藏任务：只有被触发的人能看到
    if (t.type === '隐藏') {
      var triggered = (t.claimants || []).some(function(c) { return c.name === name; });
      if (!triggered) return false;
    }
    return true;
  });
  
  // 分为可领取 / 已领取
  var available = filtered.filter(function(t) {
    return !(t.claimants || []).some(function(c) { return c.name === name; });
  });
  var claimed = filtered.filter(function(t) {
    return (t.claimants || []).some(function(c) { return c.name === name; });
  });
  
  var typeColors = { '主线': 'var(--red)', '支线': 'var(--yellow)', '共创': 'var(--green)', '隐藏': 'var(--card-purple)' };
  var typeIcons = { '主线': '🔴', '支线': '🟡', '共创': '🟢', '隐藏': '🟣' };
  
  function renderCard(t, idx) {
    var slots = t.max_slots || 0;
    var taken = (t.claimants || []).length;
    var slotsText = slots > 0 ? '👥 ' + taken + '/' + slots : '👥 不限';
    var isFull = slots > 0 && taken >= slots;
    
    return '<div class="quest-card ' + ({ '主线': 'gold', '支线': 'brown', '共创': 'green', '隐藏': 'purple' }[t.type] || 'brown') + '" style="margin-bottom:8px">' +
      '<div class="quest-card-header">' +
        '<span class="quest-card-badge">' + (typeIcons[t.type]||'') + ' ' + esc(t.type) + '</span>' +
        '<span class="quest-card-type">' + esc(t.category||'') + '</span>' +
      '</div>' +
      '<div class="quest-card-title">' + esc(t.name) + '</div>' +
      '<div class="quest-card-meta">' +
        '<span>🧙 ' + esc(t.poster||'NT管理员') + '</span>' +
        '<span>📅 ' + esc(t.deadline||'—') + '</span>' +
        '<span>' + slotsText + '</span>' +
      '</div>' +
      (t.note ? '<div style="font-size:.78rem;color:var(--muted);margin-bottom:6px">📝 ' + esc(t.note) + '</div>' : '') +
      '<div class="quest-card-reward" style="display:flex;justify-content:space-between;align-items:center">' +
        '<span>💎 NT ' + (t.points||0) + '</span>' +
        (isFull ? '<span style="color:var(--red);font-size:.8rem">已满</span>' : '<button class="btn pri" onclick="claimTask(' + idx + ')" style="font-size:.75rem;padding:4px 12px">⚔️ 领取</button>') +
      '</div></div>';
  }
  
  function renderClaimedCard(c) {
    var t = c.task; var cl = c.claim;
    var stText = cl.status === 'submitted' ? '⏳审核中' : cl.status === 'in_progress' ? '🔄进行中' : cl.status;
    return '<div style="display:flex;align-items:center;justify-content:space-between;padding:8px;background:var(--card);border:1px solid var(--border);border-radius:8px;margin-bottom:6px;opacity:.85">' +
      '<div><span style="font-weight:600">' + esc(t.name) + '</span><span style="font-size:.75rem;color:var(--muted);margin-left:8px">NT ' + (t.points||0) + ' · ' + stText + '</span></div>' +
      '<div style="display:flex;gap:6px">' +
        (cl.status === 'in_progress' ? '<button class="btn green" onclick="submitTask(' + c.idx + ')" style="font-size:.7rem;padding:2px 10px">提交</button>' : '') +
        (cl.status === 'in_progress' ? '<button class="btn" onclick="dropTask(' + c.idx + ')" style="font-size:.7rem;padding:2px 10px">放弃</button>' : '') +
      '</div></div>';
  }
  
  container.innerHTML =
    '<div class="card" style="margin-bottom:8px;padding:8px 14px;display:flex;gap:6px;flex-wrap:wrap">' +
      ['全部','主线','支线','共创','隐藏'].map(function(f) {
        return '<button class="btn' + (questFilter === f ? ' pri' : '') + '" onclick="questFilter=\'' + f + '\';renderQuestBoard()" style="font-size:.75rem;padding:4px 12px">' + (typeIcons[f]||'') + ' ' + f + '</button>';
      }).join('') +
    '</div>' +
    (available.length > 0 ? '<div class="card"><h3>🟢 可领取（' + available.length + '项）</h3>' + 
      available.map(function(t) { var idx = tasks.indexOf(t); return renderCard(t, idx); }).join('') + '</div>' : '') +
    (claimed.length > 0 ? '<div class="card"><h3>📋 已领取（' + claimed.length + '项）</h3>' + 
      claimed.map(function(c) { return renderClaimedCard(c); }).join('') + '</div>' : '') +
    (available.length === 0 && claimed.length === 0 ? '<div class="card"><div style="text-align:center;color:var(--muted);padding:16px">📭 当前没有符合条件的任务</div></div>' : '');
}
```

### 3.4 交互函数

```js
// ── 领取任务 ──
function claimTask(idx) {
  var t = data.tasks[idx];
  if (!t || !currentUser) return;
  if (!t.claimants) t.claimants = [];
  // 检查是否已领
  if (t.claimants.some(function(c) { return c.name === currentUser.name; })) {
    alert('你已经领取了这个委托'); return;
  }
  // 检查是否已满
  if (t.max_slots > 0 && t.claimants.length >= t.max_slots) {
    alert('该委托名额已满'); return;
  }
  t.claimants.push({ name: currentUser.name, claimed_at: new Date().toISOString().slice(0,10), status: 'in_progress' });
  localStorage.setItem('camp_data', JSON.stringify(data));
  renderQuestBoard();
  setStatus('已领取：' + t.name);
}

// ── 提交完成 ──
function submitTask(idx) {
  var t = data.tasks[idx];
  if (!t || !currentUser) return;
  var cl = (t.claimants || []).find(function(c) { return c.name === currentUser.name; });
  if (!cl) { alert('你还没有领取这个委托'); return; }
  var note = prompt('描述你完成的内容（可选）：');
  cl.status = 'submitted';
  cl.submitted_at = new Date().toISOString().slice(0,10);
  cl.submit_note = note || '';
  localStorage.setItem('camp_data', JSON.stringify(data));
  renderQuestBoard();
  setStatus('已提交，等待审核：' + t.name);
}

// ── 放弃委托 ──
function dropTask(idx) {
  var t = data.tasks[idx];
  if (!t || !currentUser) return;
  var cl = (t.claimants || []).find(function(c) { return c.name === currentUser.name; });
  if (!cl || cl.status !== 'in_progress') { alert('无法放弃'); return; }
  if (!confirm('确定放弃「' + t.name + '」？')) return;
  cl.status = 'dropped';
  localStorage.setItem('camp_data', JSON.stringify(data));
  renderQuestBoard();
  setStatus('已放弃：' + t.name);
}

// ── 隐藏任务触发（NT管理员操作）──
function triggerHiddenTask(taskIdx, adventurerName) {
  var t = data.tasks[taskIdx];
  if (!t || t.type !== '隐藏') return;
  if (!t.claimants) t.claimants = [];
  // 检查是否已触发
  if (t.claimants.some(function(c) { return c.name === adventurerName; })) {
    alert('已对该冒险者触发过此任务'); return;
  }
  t.claimants.push({ name: adventurerName, claimed_at: new Date().toISOString().slice(0,10), status: 'triggered' });
  localStorage.setItem('camp_data', JSON.stringify(data));
  setStatus('已触发隐藏任务「' + t.name + '」给 ' + adventurerName);
}
```

```js
function renderNTTaskPanel() {
  var tasks = data.tasks || [];

' · "' + esc(p.claim.submit_note) + '"' : '') + '</div></div><div style="display:flex;gap:6px"><button class="btn green" onclick="reviewTask(' + p.idx + ',\'' + escAttr(p.claim.name) + '\',\'approve\')" style="font-size:.7rem">✅ 通过</button><button class="btn danger" onclick="reviewTask(' + p.idx + ',\'' + escAttr(p.claim.name) + '\',\'reject\')" style="font-size:.7rem">❌ 退回</button></div></div>';

function showPostTask() {
  var name = prompt('任务名称：'); if (!name) return;
  var type = prompt('任务类型（主线/支线/共创/隐藏）：') || '支线';
  var pts = parseInt(prompt('NT 奖励：')) || 10;
  var slots = parseInt(prompt('人数限制（0=不限）：')) || 0;
  var deadline = prompt('截止日期：') || '';
  var note = prompt('任务说明：') || '';
  var criteria = prompt('验收标准：') || '';
  
  data.tasks.push({
    name: name, type: type, category: '课程', points: pts,
    max_slots: slots, deadline: deadline, source: '第四期NT任务系统',
    poster: currentUser.name, posted_at: new Date().toISOString().slice(0,10),
    claimants: [], status: 'active', note: note, accept_criteria: criteria
  });
  localStorage.setItem('camp_data', JSON.stringify(data));
  renderNTTaskPanel();
  setStatus('已发布：' + name);
}

function reviewTask(taskIdx, name, action) {
  var t = data.tasks[taskIdx];
  var cl = (t.claimants||[]).find(function(c) { return c.name === name; });
  if (!cl) return;
  
  if (action === 'approve') {
    cl.status = 'completed';
    cl.reviewed_by = currentUser.name;
    cl.reviewed_at = new Date().toISOString().slice(0,10);
    
    // NT 池检查
    var budget = data.budget || {};
    var poolTotal = budget.nt_total_pool || 30000;
    var totalIssued = (data.finance || []).filter(function(tx) { return tx.type === 'task_reward'; }).reduce(function(s, tx) { return s + (tx.amount||0); }, 0);
    if (totalIssued + (t.points||0) > poolTotal) { alert('⚠️ NT 池余额不足！'); return; }
    
    // 写流水
    if (!data.finance) data.finance = [];
    data.finance.push({
      type: 'task_reward', from: '系统', to: name,
      amount: t.points||0, task_name: t.name, task_id: taskIdx,
      date: new Date().toISOString().slice(0,10), confirmed_by: currentUser.name
    });
  } else {
    cl.status = 'in_progress';
    var reason = prompt('退回原因（可选）：');
    cl.review_note = reason || '';
  }
  localStorage.setItem('camp_data', JSON.stringify(data));
  renderNTTaskPanel();
  setStatus(action === 'approve' ? '已确认：' + t.name : '已退回：' + t.name);
}

function closeTask(idx) {
  if (!confirm('关闭任务「' + data.tasks[idx].name + '」？已有领取记录将保留。')) return;
  data.tasks[idx].status = 'closed';
  localStorage.setItem('camp_data', JSON.stringify(data));
  renderNTTaskPanel();
}
```

```js
function saveMember() {
  var oldName = document.getElementById('editMemberOldName').value;
  var newName = document.getElementById('memberName').value.trim();
  if (!newName) return alert('姓名不能为空');

  // ── 1. 更新 data.members ──
  if (!data.members) data.members = {};
  if (oldName !== newName) {
    data.members[newName] = data.members[oldName] || {};
    delete data.members[oldName];
  }
  var mi = data.members[newName] || {};
  mi.name = newName;
  mi.gender = document.getElementById('memberGender').value;
  mi.role = document.getElementById('memberRole').value;
  mi.bio = document.getElementById('memberBio').value;
  mi.avatar_seed = editingAvatarSeed;
  mi.nt_actual = parseInt(document.getElementById('memberNT').value) || 0;
  data.members[newName] = mi;

  // ── 2. 改名时同步 data.tasks ──
  if (oldName && oldName !== newName) {
    (data.tasks || []).forEach(function(t) {
      if (t.assignee === oldName) t.assignee = newName;
      if (t.confirmer === oldName) t.confirmer = newName;
    });
    // 同步 data.finance_cny.transactions
    if (data.finance_cny && data.finance_cny.transactions) {
      data.finance_cny.transactions.forEach(function(tx) {
        if (tx.person === oldName) tx.person = newName;
      });
    }
    // 重命名 data.finance_cny.by_person 的 key
    if (data.finance_cny && data.finance_cny.by_person && data.finance_cny.by_person[oldName]) {
      data.finance_cny.by_person[newName] = data.finance_cny.by_person[oldName];
      delete data.finance_cny.by_person[oldName];
    }
    // 重命名 data.finance 中的引用
    (data.finance || []).forEach(function(tx) {
      if (tx.to === oldName) tx.to = newName;
      if (tx.from === oldName) tx.from = newName;
    });
    // 重命名 data.staff_cards 中的引用
    (data.staff_cards || []).forEach(function(c) {
      if (c.name === oldName) c.name = newName;
    });
  }

  // ── 3. 更新 nt_users ──
  var users = getUsers();
  if (users[oldName]) {
    users[newName] = users[oldName];
    users[newName].name = newName;
    users[newName].avatar_seed = editingAvatarSeed;
    if (oldName !== newName) delete users[oldName];

    var newPwd = document.getElementById('memberNewPassword').value;
```

```js
function deleteMember() {

  });

  if (data.staff_cards) {
    data.staff_cards = data.staff_cards.filter(function(c) { return c.name !== name; });

  // ── 清理 data.finance_cny ──
  if (data.finance_cny) {
    if (data.finance_cny.transactions) {
      data.finance_cny.transactions = data.finance_cny.transactions.filter(function(tx) { return tx.person !== name; });

  // ── 保留 nt_users（不删账号）──
```

```js
function getAchievements(userName) {
  var tasks = data.tasks || [];
  var completed = [];
  tasks.forEach(function(t) {
    var c = (t.claimants || []).find(function(cl) { return cl.name === userName; });
    if (c && c.status === 'completed') completed.push(t);
  });
  
  var allMain = tasks.filter(function(t) { return t.type === '主线'; });
  var doneMain = completed.filter(function(t) { return t.type === '主线'; });
  var achievements = [];
  
  if (allMain.length > 0 && doneMain.length === allMain.length) {
    achievements.push({ name: '主线通关', icon: '🏅', desc: '完成全部主线任务' });
  }
  if (completed.length >= 15) {
    achievements.push({ name: '全勤猎人', icon: '🎖️', desc: '完成 15 项以上任务' });
  }
  var doneSide = completed.filter(function(t) { return t.type === '支线'; }).length;
  if (doneSide >= 6) {
    achievements.push({ name: '支线达人', icon: '⭐', desc: '完成 6 项以上支线任务' });
  }
  
  var manual = (data.members && data.members[userName] && data.members[userName].achievements) || [];
  return achievements.concat(manual);
}
```
