function calcBudgetItemAmt(item, p) {
  var rmb = item.rmb || 0, nt = item.nt || 0;
  switch (item.calcType) {
    case 'per_day_person':
      if (item.name === '住宿费') { rmb = p.lodgingRMB; nt = p.lodgingNT; }
      if (item.name === '餐费') { rmb = p.mealRMB; nt = p.mealNT; }
      return { rmb: p.participantCount * p.duration * rmb, nt: p.participantCount * p.duration * nt };
    case 'per_person_all':
      return { rmb: p.totalPeople * rmb, nt: p.totalPeople * nt };
    default:
      return { rmb: rmb, nt: nt };
  }
}
