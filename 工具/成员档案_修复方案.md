## 一、数据结构约定

### 1.0 `nt_users.role` 与 `data.members.role` 分工

本系统有两套 role 体系，**不可混淆**：

| 字段 | 取值 | 用途 |
|------|------|------|
| `nt_users[name].role` | `'builder'` / `'adventurer'` / `'npc'` / `'visitor'` | 身份筛选（二级导航）、登录权限控制 |
| `data.members[name].role` | 中文自由文本，如 `'统筹'` / `'宣传'` / `'核心成员'` | 在档案详情中显示 |

- `filterMembers(type)` 筛选依据：`nt_users[name].role`
- 成员详情展示：`data.members[name].role`（中文显示角色名）

### 2.2 恢复并增强 `saveMember()`——同步 tasks / nt_users / finance + 密码修改

> ⚠️ 描述修正：当前代码中 `saveMember()` **不存在**（备份版第 1721-1745 行有完整实现），本次是**恢复并增强**，不是"改造"。

```js
function saveMember() {
  var oldName = document.getElementById('editMemberOldName').value;
  var newName = document.getElementById('memberName').value.trim();
  if (!newName) { alert('请输入姓名'); return; }
  if (!data.members) data.members = {};

  // ── 1. 改名时同步 data.tasks（恢复自备份版第 1728 行）──
  if (oldName && oldName !== newName) {
    data.tasks.forEach(function(t) {
      if (t.assignee === oldName) t.assignee = newName;
      if (t.confirmer === oldName) t.confirmer = newName;
    });
    delete data.members[oldName];
  }

  // ── 2. 改名时同步 data.finance_cny.transactions（新增增强）──
  if (oldName && oldName !== newName && data.finance_cny && data.finance_cny.transactions) {
    data.finance_cny.transactions.forEach(function(tx) {
      if (tx.person === oldName) tx.person = newName;
    });
    recalcFinanceByPerson();  // 重建 by_person / by_type
  }

  // ── 3. 更新 data.members ──
  data.members[newName] = {
    name: newName,
    gender: document.getElementById('memberGender').value,
    role: document.getElementById('memberRole').value.trim(),
    bio: document.getElementById('memberBio').value.trim(),
    avatar_style: editingAvatarStyle,
    avatar_seed: editingAvatarSeed,
    nt_actual: parseInt(document.getElementById('memberNT').value) || 0,
  };

  // ── 4. 同步 nt_users（新增增强）──
  var users = getUsers();
  if (users[oldName]) {
    users[newName] = users[oldName];
    users[newName].name = newName;
    users[newName].avatar_seed = editingAvatarSeed;
    if (oldName !== newName) delete users[oldName];
  } else if (!users[newName]) {
    // 人在 data.members 里但没有 nt_users 记录 → 自动创建
    users[newName] = {
      name: newName,
      role: 'adventurer',
      password: null,
      avatar_seed: editingAvatarSeed,
      created: new Date().toISOString().slice(0, 10)
    };
  } else {
    users[newName].avatar_seed = editingAvatarSeed;
  }

  // ── 5. 密码修改（新增增强）──
  var newPwd = document.getElementById('memberNewPassword').value;
  if (newPwd) {
    var confirmPwd = document.getElementById('memberConfirmPassword').value;
    if (newPwd !== confirmPwd) return alert('两次密码不一致');
    if (newPwd.length < 2) return alert('密码至少 2 位');
    users[newName].password = encodePassword(newPwd, newName);
  }

  saveUsers(users);
  localStorage.setItem('camp_data', JSON.stringify(data));
  document.getElementById('memberModal').classList.remove('show');
  renderMembers();
  setStatus('已保存：' + newName);
}
```

### 2.3 恢复并增强 `deleteMember()` 函数

> 当前第 1104 行 `<button onclick="deleteMember()">` 引用了不存在的函数。备份版第 1747-1759 行有完整实现。

```js
function deleteMember() {
  var name = document.getElementById('editMemberOldName').value;
  if (!name || !confirm('确定删除成员「' + name + '」？任务不会被删除，但会标记为未分配。')) return;

  // ── 1. 清除 data.tasks 中的分配（恢复自备份版）──
  data.tasks.forEach(function(t) {
    if (t.assignee === name) t.assignee = '';
    if (t.confirmer === name) t.confirmer = '';
  });

  // ── 2. 删除 data.members（恢复自备份版）──
  delete data.members[name];

  // ── 3. 清理 data.finance_cny.by_person（恢复自备份版）──
  if (data.finance_cny && data.finance_cny.by_person && data.finance_cny.by_person[name]) {
    delete data.finance_cny.by_person[name];
  }

  // ── 4. 清理 data.finance_cny.transactions 中的 person 引用（新增增强）──
  if (data.finance_cny && data.finance_cny.transactions) {
    data.finance_cny.transactions.forEach(function(tx) {
      if (tx.person === name) tx.person = '';
    });
    recalcFinanceByPerson();
  }

  // ── 5. 同步删除 nt_users（新增增强）──
  var users = getUsers();
  if (users[name]) {
    delete users[name];
    saveUsers(users);
  }

  document.getElementById('memberModal').classList.remove('show');
  localStorage.setItem('camp_data', JSON.stringify(data));
  renderMembers();
  setStatus('已删除：' + name);
}
```

### 2.6 改造 `renderMembers()`——按身份筛选 + 数据合并

**筛选逻辑**：

```js
var memberFilter = 'builder';  // 默认显示共建者

function filterMembers(type) {
  memberFilter = type;
  // 更新按钮高亮
  var types = ['builder','adventurer','npc','visitor'];
  document.querySelectorAll('#memberSubNav .btn').forEach(function(b, i) {
    b.className = (types[i] === type) ? 'btn pri' : 'btn';
  });
  renderMembers();
}
```

**数据合并逻辑**（`renderMembers()` 改造核心）：

当前 `renderMembers()` 只从 `data.tasks` 取人 → 改为**从 `nt_users` 取所有该类型用户**，再从 `data.tasks` 和 `data.members` 补充任务和材料信息。

三种数据来源的合并规则：

| 情况 | 处理 |
|------|------|
| 人在 `nt_users` 且 `data.members` 也有 | 合并显示，`data.members` 字段（role、bio 等）优先 |
| 人在 `nt_users` 但不在 `data.members` | 用默认值创建显示：role=''，bio=''，nt_actual=0 |
| 人在 `data.members` 但不在 `nt_users` | `migrateExistingMembers()` 已自动补充到 `nt_users`（idempotent），取合并后结果 |

伪代码：

```js
function renderMembers() {
  var users = getUsers();
  var tasks = data.tasks || [];
  var members = data.members || {};

  staffData = {};

  // 1. 从 nt_users 取所有该身份的用户
  Object.keys(users).forEach(function(name) {
    var user = users[name];
    if (user.role !== memberFilter) return;  // 按二级导航筛选
    if (!staffData[name]) staffData[name] = { tasks: [], points: 0, done: 0, active: 0, todo: 0 };
  });

  // 2. 从 data.tasks 补充任务统计
  tasks.forEach(function(t) {
    var name = t.assignee || '(未分配)';
    if (!staffData[name]) return;  // 不在当前筛选范围内，跳过
    staffData[name].tasks.push(t);
    staffData[name].points += (t.points || 1);
    if (t.status === '已完成' || t.status === '已确认') staffData[name].done++;
    else if (t.status === '进行中') staffData[name].active++;
    else staffData[name].todo++;
  });

  // 3. 按任务数排序
  var sorted = Object.entries(staffData).sort(function(a, b) {
    return b[1].tasks.length - a[1].tasks.length;
  });

  // ...
}
```
