### 图标匹配规则

```js
function getFoodIcon(name) {
  var map = {
    '饭': '🍚', '米': '🍚', '面': '🍜', '汤': '🍲',
    '豆腐': '🫘', '豆': '🫘', '菜': '🥬', '蔬': '🥬',
    '瓜': '🥒', '茄': '🍆', '薯': '🍠', '花生': '🥜',
    '茶': '🍵', '水': '💧', '蛋': '🥚', '菇': '🍄',
    '饺子': '🥟', '包': '🥟', '沙拉': '🥗', '果': '🍎'
  };
  for (var k in map) { if (name.indexOf(k) !== -1) return map[k]; }
  return '🍽️'; // default
}
```

```
  ┌──────────────────────────────┐

```

```
  → 物品状态 → 已售
  → 双方流水记录
  → 卖家信箱通知
  → 线下交付
```

```js
function computeMealReminders() {
  var msgs = [];

  var hour = now.getHours();
  var today = now.toISOString().slice(0, 10);

  // 午餐预定提醒 (10:00-10:30)
  if (hour === 10) {
    var ordered = (data.canteen_orders || []).some(function(o) {
```
