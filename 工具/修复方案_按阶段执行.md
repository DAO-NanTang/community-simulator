## 〇、前置工作：提取共用函数

**在执行任何 Step 之前**，先建立两个共用函数。后续所有 Step 都基于它们。

### 共用函数 A：任务数据合并

在 `dashboard.html` 的 `<script>` 中，找一个合适位置（建议放在 data 初始化之后、第一个 render 函数之前），插入以下代码：

```js
// ═══ 共用：wizard → data.tasks 字段保护 ═══
// 创世向导保存时不会覆盖的 NT 系统字段列表
var NT_MERGE_FIELDS = ['scope','visibility','reviewer','min_season',
  'claimants','max_slots','poster','posted_at','materials','subtasks','outputs','steps'];

// 将 wizardTasks 安全合并到 data.tasks，保留 NT 系统字段
function mergeWizardToData(wizardTasks) {
  var existingByName = {};
  (data.tasks || []).forEach(function(t) {
    if (t.name) existingByName[t.name] = t;
  });
  data.tasks = wizardTasks.map(function(t) {
    var existing = existingByName[t.name];
    if (existing) {
      NT_MERGE_FIELDS.forEach(function(f) {
        if (existing[f] !== undefined) t[f] = existing[f];
      });
    } else {
      // 新任务：给 NT 系统字段设默认值
      if (t.scope === undefined) t.scope = 'camp';
      if (t.visibility === undefined) t.visibility = 'all';
      if (t.reviewer === undefined) t.reviewer = '';
      if (t.min_season === undefined) t.min_season = 0;
      if (!t.claimants) t.claimants = [];
      if (t.max_slots === undefined) t.max_slots = 0;
      if (!t.poster) t.poster = '';
      if (!t.posted_at) t.posted_at = '';
    }
    return Object.assign({}, t);
  });
  localStorage.setItem('camp_data', JSON.stringify(data));
}
```

### 共用函数 B：NT 余额计算

```js
// ═══ 共用：按 scope 计算 NT 余额 ═══
// 返回 {camp: number, personal: number}
function calcNtByScope(name) {
  var camp = 0, personal = 0;
  (data.finance || []).forEach(function(tx) {
    var scope = tx.scope || 'camp';
    var amt = tx.amount || 0;
    if (tx.to === name) {
      if (scope === 'personal') personal += amt; else camp += amt;
    }
    if (tx.from === name) {
      if (scope === 'personal') personal -= amt; else camp -= amt;
    }
  });
  return { camp: camp, personal: personal };
}

// 计算某人总 NT（不区分 scope）
function calcNtTotal(name) {
  var r = calcNtByScope(name);
  return r.camp + r.personal;
}
```

### 共用函数 C：财务记录清理

```js
// ═══ 共用：从 data.finance 中移除关联记录 ═══
// filterType: 'task_name' | 'person'（按 to 或 from 匹配）
function removeFromFinance(filterType, value) {
  data.finance = (data.finance || []).filter(function(tx) {
    if (filterType === 'task_name') return tx.task_name !== value;
    if (filterType === 'person') return tx.to !== value && tx.from !== value;
    return true;
  });
}
```
