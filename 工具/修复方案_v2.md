```javascript
function getAnchorSet(dates) {
  // 锚点列索引（以 dates 长度为准）
  // 注意：index 0 = 报到日，不标记为锚点
  var n = dates.length;
  var photoDay = Math.floor(n / 2);      // 合影日 = 营期中点
  var exhDay = n - 2;                     // 展览日 = 结营前2天
  var lastDay = n - 1;                    // 结营日
  var anchorSet = {};
  // 开营日 index 1，展览相关列，合影日，结营日
  [1, photoDay, exhDay-3, exhDay-2, exhDay-1, exhDay, lastDay].forEach(function(d) {
    if (d >= 0) anchorSet[d] = true;
  });
  return anchorSet;
}
```

然后 `renderDailySchedule()` 和 `colDrop()` 都调用此函数。

## 第3项: 里程碑从活动日行内容读取锚点

### 问题

`generateMilestones()` 有三处硬编码：
1. `schedDates[1]` = 开营日（假设index 1）
2. 合影日/展览日用长度公式计算
3. 覆盖 `data.camp_dates.start`（与用户初衷不符，但审查指出应覆盖，因为日历是数据源头）

### 修订：遵循 锚点修正方案_v2.md

审查指出应**遵循锚点修正方案_v2.md**：覆盖 `camp_dates.start`。理由：日历表是经过确认的数据源头，`camp_dates.start` 是派生值。不一致时以日历为准。

### 修改

#### A. 新增 `findAnchorDates()` 函数

**审查指出的问题：** 子串匹配 `indexOf('开营')` 会误匹配「开营前动员会」。

**修正：** 用 `🔒` 前缀做精确匹配。模板填充的锚点内容格式为 `🔒 开营第一天`：

```javascript
function findAnchorDates() {
  var dates = (data.schedule && data.schedule.dates) ? data.schedule.dates : [];
  if (dates.length === 0) return null;

  // 找 "🎯 活动日" 行（section 含"活动日"）
  var activitySlot = (data.schedule.slots || []).find(function(s) {
    return s.section && s.section.indexOf('活动日') >= 0;
  });
  if (!activitySlot || !activitySlot.activities) return null;

  var anchors = {};
  activitySlot.activities.forEach(function(cell, i) {
    if (!cell || i >= dates.length) return;
    // 用 🔒 前缀做精确锚定
    if (!anchors.openDay && /🔒.*开营/.test(cell))
      anchors.openDay = { index:i, date:dates[i] };
    if (!anchors.closeDay && /🔒.*结营/.test(cell))
      anchors.closeDay = { index:i, date:dates[i] };
    if (!anchors.exhDay && /🔒.*展览/.test(cell))
      anchors.exhDay = { index:i, date:dates[i] };
    if (!anchors.photoDay && /🔒.*合影/.test(cell))
      anchors.photoDay = { index:i, date:dates[i] };
  });
  return anchors;
}
```

#### B. 修改 `generateMilestones()`

```javascript
function generateMilestones() {
  var schedDates = (data.schedule && data.schedule.dates) ? data.schedule.dates : [];
  if (schedDates.length < 2) { /* 错误处理 */ return; }

  var anchors = findAnchorDates();

  // 确定开营日：优先从活动日行读取，fallback 到 schedDates[1]
  var openMD, closeMD;
  if (anchors && anchors.openDay) {
    openMD = anchors.openDay.date;
  } else {
    openMD = schedDates[1]; // fallback: index 1 = 开营日
  }
  if (anchors && anchors.closeDay) {
    closeMD = anchors.closeDay.date;
  } else {
    closeMD = schedDates[schedDates.length - 1]; // fallback: 最后一天
  }

  var openParts = openMD.split('/');
  var closeParts = closeMD.split('/');
  var y = new Date().getFullYear();
  var startD = new Date(y, parseInt(openParts[0])-1, parseInt(openParts[1]));
  var endD = new Date(y, parseInt(closeParts[0])-1, parseInt(closeParts[1]));
  var campDays = Math.round((endD - startD) / 86400000);

  // 遵循 锚点修正方案_v2.md：覆盖 camp_dates（日历是数据源头）
  if (!data.camp_dates) data.camp_dates = {};
  data.camp_dates.start = startD.toISOString().slice(0,10);
  data.camp_dates.end = endD.toISOString().slice(0,10);
  data.camp_dates.duration_days = campDays;
  var wi = document.getElementById('wizStartDate'); if (wi) wi.value = data.camp_dates.start;
  var wd = document.getElementById('wizDuration'); if (wd) wd.value = campDays;

  // 生成里程碑
  var ms = [];
  PRESET_MILESTONES.forEach(function(pm) {
    var d;
    // 展览相关偏移用展览日锚点
    if (anchors && anchors.exhDay && typeof pm.offset === 'string' && pm.offset.indexOf('ex') >= 0) {
      var exhParts = anchors.exhDay.date.split('/');
      var exhD = new Date(y, parseInt(exhParts[0])-1, parseInt(exhParts[1]));
      var offsetNum = parseInt(pm.offset.replace(/[^0-9-]/g, '')) || 0;
      exhD.setDate(exhD.getDate() + offsetNum);
      d = exhD;
    } else {
      d = calcMilestoneDate(startD, endD, campDays, pm.offset);
    }
    var existing = (data.camp_dates.milestones || []).find(function(m) { return m.id === pm.id; });
    var isLocked = (existing && existing._userLocked) ? true : pm.locked;
    ms.push({
      id:pm.id, name:pm.name, icon:pm.icon, offset:pm.offset,
      date:d.toISOString().slice(0,10), locked:isLocked
    });
  });
  data.camp_dates.milestones = ms;
  localStorage.setItem('camp_data', JSON.stringify(data));
  recalcMilestonesUI();
}
```
