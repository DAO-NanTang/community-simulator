```javascript
function importTasksFromExisting(pool) {
  if (!confirm('从已有任务导入？当前任务池的内容将被替换。')) return;
  var isStaff = pool === 'staff';
  var staffNames = ['砚仁','朝林','小红','若曦','淑惠','跳大爷','fc'];
  var filtered = (data.tasks || []).filter(function(t) {
    var name = t.assignee || '';
    return isStaff ? staffNames.some(function(s) { return name.indexOf(s) !== -1; }) || !name
                    : name && !staffNames.some(function(s) { return name.indexOf(s) !== -1; });
  });
  if (pool === 'staff') settingStaffTasks = filtered.map(function(t) { return Object.assign({}, t); });
  else settingCreatorTasks = filtered.map(function(t) { return Object.assign({}, t); });
  renderSettingTasks(pool);
}
```

```javascript
function commitSettingTasks(pool) {
  var arr = getSettingTasks(pool);
  if (arr.length === 0) { alert('没有可提交的任务'); return; }
  if (!confirm('将 ' + arr.length + ' 个任务提交到任务大厅？')) return;
  arr.forEach(function(st) {
    var existing = data.tasks.findIndex(function(t) { return t.name === st.name; });
    if (existing >= 0) {
      data.tasks[existing] = st;
    } else {
      data.tasks.push(Object.assign({}, st));
    }
  });
  // 清空当前池
  if (pool === 'staff') settingStaffTasks = []; else settingCreatorTasks = [];
  updateAll();
  setStatus('已提交 ' + arr.length + ' 个任务到任务大厅');
  renderSettingTasks(pool);
}
```

```javascript
// ── 模板管理函数 ──

function loadTemplates() {
  var stored = localStorage.getItem(TEMPLATE_STORAGE_KEY);
  if (stored) {
    try { return JSON.parse(stored); } catch(e) {}
  }
  // 首次使用：写入预设模板
  var defaults = JSON.parse(JSON.stringify(PRESET_TEMPLATES));
  localStorage.setItem(TEMPLATE_STORAGE_KEY, JSON.stringify(defaults));
  return defaults;
}

function getTemplatesByPool(pool) {
  var templates = loadTemplates();
  return Object.entries(templates).filter(function(e) {
    return e[1].pool === pool;
  });
}

function refreshTemplateSelects() {
  ['staff','creator'].forEach(function(pool) {
    var selId = pool === 'staff' ? 'staffTemplateSelect' : 'creatorTemplateSelect';
    var sel = document.getElementById(selId);
    if (!sel) return;
    var templates = getTemplatesByPool(pool);
    sel.innerHTML = '<option value="">-- 选择模板 --</option>' +
      templates.map(function(e) {
        var name = e[0], t = e[1];
        var isPreset = PRESET_TEMPLATES.hasOwnProperty(name);
        var icon = isPreset ? '🔒' : '📝';
        return '<option value="' + escAttr(name) + '">' + icon + ' ' + esc(name) + ' (' + t.tasks.length + '项)</option>';
      }).join('');
  });
}

function applyTemplate(pool) {
  var selId = pool === 'staff' ? 'staffTemplateSelect' : 'creatorTemplateSelect';
  var name = document.getElementById(selId).value;
  if (!name) { alert('请先选择一个模板'); return; }
  var arr = getSettingTasks(pool);
  if (arr.length > 0 && !confirm('当前表格内容将被替换，是否继续？')) return;
  var templates = loadTemplates();
  var template = templates[name];
  if (!template) { alert('模板不存在'); return; }
  // 深拷贝任务数据
  if (pool === 'staff') {
    settingStaffTasks = template.tasks.map(function(t) { return Object.assign({}, t); });
  } else {
    settingCreatorTasks = template.tasks.map(function(t) { return Object.assign({}, t); });
  }
  renderSettingTasks(pool);
  setStatus('已应用模板: ' + name);
}

function saveTemplate(pool) {
  var arr = getSettingTasks(pool);
  if (arr.length === 0) { alert('当前表格为空，无法保存模板'); return; }
  var name = prompt('模板名称：');
  if (!name) return;
  var templates = loadTemplates();
  if (templates[name] && !confirm('模板「' + name + '」已存在，是否覆盖？')) return;
  templates[name] = {
    pool: pool,
    tasks: arr.map(function(t) { return Object.assign({}, t); })
  };
  localStorage.setItem(TEMPLATE_STORAGE_KEY, JSON.stringify(templates));
  refreshTemplateSelects();
  setStatus('模板已保存: ' + name);
}

function deleteTemplate(pool) {
  var selId = pool === 'staff' ? 'staffTemplateSelect' : 'creatorTemplateSelect';
  var name = document.getElementById(selId).value;
  if (!name) { alert('请先选择要删除的模板'); return; }
  if (PRESET_TEMPLATES.hasOwnProperty(name)) {
    alert('预设模板不可删除。如需恢复原始内容，请在⑥ AI导入页点击「恢复默认模板」。');
    return;
  }
  if (!confirm('确定删除模板「' + name + '」？此操作不可恢复。')) return;
  var templates = loadTemplates();
  delete templates[name];
  localStorage.setItem(TEMPLATE_STORAGE_KEY, JSON.stringify(templates));
  refreshTemplateSelects();
  setStatus('已删除模板: ' + name);
}

function restoreDefaultTemplates() {
  if (!confirm('恢复全部 10 个预设模板到默认状态？\n（您自建的模板不会被删除，同名预设模板将被覆盖）')) return;
  var templates = loadTemplates();
  Object.keys(PRESET_TEMPLATES).forEach(function(name) {
    templates[name] = JSON.parse(JSON.stringify(PRESET_TEMPLATES[name]));
  });
  localStorage.setItem(TEMPLATE_STORAGE_KEY, JSON.stringify(templates));
  refreshTemplateSelects();
  setStatus('已恢复 ' + Object.keys(PRESET_TEMPLATES).length + ' 个预设模板');
}

// ── 初始化：确保模板已加载 ──
document.addEventListener('DOMContentLoaded', function() {
  loadTemplates(); // 首次触发自动写入预设
  refreshTemplateSelects();
});
```
